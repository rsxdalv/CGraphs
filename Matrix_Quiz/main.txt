#include <iostream>

class virsotnes
{
	public:
		virsotnes();

};

virsotnes::virsotnes()
{
	bijis = false;
}


bool rek(int gatavs[][5], int v1, int v2, int from)
{
	bool cikls=false;
	for(int next=0; next<5; next++)
    {
		if((gatavs[v1][next]>0)&&(from!=next))
		{
			if(next==v2)
			{
				cikls=true;
				//cout << v1 << "  " << v2 << "  " << next << endl;
                //system("PAUSE");
			}
			else
			{
                //cout << v1 << "  " << v2 << "  " << next << endl;
                
                //system("PAUSE");
				rek(gatavs,  next, v2, v1);
			}
		}
		//cout << next << endl;
		//system("PAUSE");
	}
	if(cikls==false)
	{
		cout << v1 << "  " << v2 << endl;
		system("PAUSE");
	}
	return cikls;
}
